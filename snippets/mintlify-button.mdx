import React from 'react';

export function MintlifyButton({ children, href, onClick }) {
  const baseClasses =
    'inline-flex items-center justify-center px-4 py-2 text-sm font-medium rounded-md transition';
  const lightClasses =
    'text-zinc-900 bg-zinc-100 hover:bg-zinc-200';
  const darkClasses =
    'dark:text-zinc-100 dark:bg-zinc-800 dark:hover:bg-zinc-700';

  const className = `${baseClasses} ${lightClasses} ${darkClasses}`;

  return href ? (
    <a href={href} className={className}>
      {children}
    </a>
  ) : (
    <button onClick={onClick} className={className}>
      {children}
    </button>
  );
}
