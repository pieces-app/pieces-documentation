export function ButtonGroup({ cols = 3, gap = 1, children }) {
  const [mounted, setMounted] = useState(false);
  const [colsCount, setColsCount] = useState(cols);

  useEffect(() => {
    setMounted(true);
    function update() {
      const w = window.innerWidth;
      if (w < 640) setColsCount(1);
      else if (w < 768) setColsCount(Math.min(cols, 2));
      else setColsCount(cols);
    }
    update();
    window.addEventListener('resize', update);
    return () => window.removeEventListener('resize', update);
  }, [cols]);

  if (!mounted) return null;

  return (
    <div
      style={{
        display: 'grid',
        gridTemplateColumns: `repeat(${colsCount}, minmax(0, 1fr))`,
        gap: `${gap}rem`,
      }}
    >
      {children}
    </div>
  );
}

export function IconButton({ icon, text, href, newTab = false, size = 'md' }) {
  const [mounted, setMounted] = useState(false);
  const [isDark, setIsDark] = useState(false);

  useEffect(() => {
    setMounted(true);
    const html = document.documentElement;
    const updateDark = () => setIsDark(html.classList.contains('dark'));
    updateDark();
    const mo = new MutationObserver(updateDark);
    mo.observe(html, { attributes: true, attributeFilter: ['class'] });
    return () => mo.disconnect();
  }, []);

  if (!mounted) return null;

  const bg       = isDark ? '#0e0e0f'             : '#ffffff';
  const border   = isDark ? 'rgba(255,236,209,0.1)' : '#D1D5DB';
  const hoverCol = isDark ? 'rgba(255,236,209,1)'   : '#4B5563';
  const arrowCol = isDark ? 'rgba(255,236,209,0.4)' : '#6B7280';
  const sizeCls  =
    size === 'sm' ? 'px-4 py-3 text-sm' :
    size === 'lg' ? 'px-8 py-7 text-lg' :
                    'px-6 py-5 text-base';

  const onEnter = e => {
    e.currentTarget.style.border = `1px solid ${hoverCol}`;
    e.currentTarget.querySelector('svg')?.setAttribute('stroke', hoverCol);
  };
  const onLeave = e => {
    e.currentTarget.style.border = `1px solid ${border}`;
    e.currentTarget.querySelector('svg')?.setAttribute('stroke', arrowCol);
  };

  return (
    <a
      href={href}
      target={newTab ? '_blank' : '_self'}
      rel={newTab ? 'noopener noreferrer' : undefined}
      className={`${sizeCls} relative flex items-center gap-x-3 w-full rounded-xl`}
      style={{
        backgroundColor: bg,
        border:           `1px solid ${border}`,
        transition:       'border-color 150ms ease-in-out',
      }}
      onMouseEnter={onEnter}
      onMouseLeave={onLeave}
    >
      <div className="absolute top-5 right-5">
        <svg
          className="w-4 h-4"
          xmlns="http://www.w3.org/2000/svg"
          viewBox="0 0 24 24"
          fill="none"
          stroke={arrowCol}
          strokeWidth={2}
          strokeLinecap="round"
          strokeLinejoin="round"
          style={{ transition: 'stroke 150ms ease-in-out' }}
        >
          <path d="M7 7h10v10" />
          <path d="M7 17L17 7" />
        </svg>
      </div>
      <div className="h-6 w-6">
        <img src={icon} alt="" className="h-full w-full object-contain" />
      </div>
      <span className="font-semibold text-base">{text}</span>
    </a>
  );
}
