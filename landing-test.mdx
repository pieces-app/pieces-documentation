export function ColorCopyButton() {
  const [hue, setHue] = useState(180)
  const [saturation, setSaturation] = useState(50)
  const [lightness, setLightness] = useState(50)
  const [baseColor, setBaseColor] = useState(`hsl(${hue}, ${saturation}%, ${lightness}%)`)

  // Recompute baseColor whenever any value changes
  useEffect(() => {
    setBaseColor(`hsl(${hue}, ${saturation}%, ${lightness}%)`)
  }, [hue, saturation, lightness])

  // copy baseColor on click
  const copyToClipboard = () => {
    navigator.clipboard
      .writeText(baseColor)
      .then(() => alert(`Copied ${baseColor}`))
      .catch(err => console.error('Copy failed:', err))
  }

  return (
    <button
      onClick={copyToClipboard}
      style={{
        padding: '0.5rem 1rem',
        backgroundColor: baseColor,
        color: '#fff',
        border: 'none',
        borderRadius: '4px',
        cursor: 'pointer',
      }}
      title={`Copy ${baseColor}`}
    >
      Copy Base Color ({baseColor})
    </button>
  )
}
